def self.reset(*properties)
  reset_variables = properties.empty? ? instance_variables : instance_variables.map { |p| p.to_s} & \
                                                             properties.map         { |p| "@#{p}" }
  reset_variables.each { |v| instance_variable_set(v.to_sym, nil) }
end
